
model {
    alpha0 ~ dnorm(0, 0.1)
    logit(p0) <- alpha0
    alpha1 ~ dnorm(0, 0.1)
    sigma <- pow(1/(2*alpha1), 0.5) # circular home range?
    psi ~ dunif(0, 1)
    for(i in 1:M) {
        z[i] ~ dbern(psi)
        s[i] ~ dunif(xlimA[1], xlimA[2])
        for(j in 1:n_traps) {
            d[i,j] <- abs(s[i] - traplocsA[j])
            y[i,j] ~ dbin(p[i,j], K)
            p[i,j] <- z[i]*p0*exp(- alpha1 * d[i,j] * d[i,j])
      }
    }

N <- sum(z[])
D <- N/(xlimA[2] - xlimA[1])
}
